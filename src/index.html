Api Rest GAS

Api Rest usada para cruds con bases de datos de tipo SpreadSheet en Google

--------------------------------------------------------------------------------

Metodos soportados 

        
metodos get por query string


        
buscar todo en una sheet:

        interface findAllByNamesheet {
        nameSheet: string;
        }
        
buscar datos por un id:

        interface findDataById {
        nameSheet: string;
        id: string;
        nameId: string;
        }
        
        interface findMultipleDataById {
        nameSheet: string;
        id: string;
        nameId: string;
        }
        
        interface saveObjectInSheet {
        datos: {};
        nameSheet: string;
        }
        
        interface updateObjectInSheet {
        id: string;
        datos: {};
        nameSheet: string;
        nameId: string;
        }
        
        interface deleteObjectInSheet {
        id: string;
        nameSheet: string;
        nameId: string;
        }
        
        interface guardarCsv {
        data: string;
        nameSheet: string;
        }
        
        interface saveFileInDrive {
        mimeType: string;
        bytes: any[];
        filename: string;
        }
        
        interface returnApiFind {
        status: string;
        message: string;
        data: any[];
        }
